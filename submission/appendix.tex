\subsection*{C.1 Priority-Based Decision Logic}

\begin{lstlisting}[language=Python, caption={Decision Agent Core Logic}]
def decide(self, trend_result, risk_result, 
           forecast_result):
    trend_score = trend_result['trend_score']
    risk_score = risk_result['risk_score']
    forecast_change = forecast_result['percentChange']
    
    # P1: Critical Risk - Safety First
    if risk_score >= 8:
        return {
            'decision': 'SELL',
            'confidence_score': 95,
            'decision_type': 'rule-based',
            'rule_triggered': 'P1',
            'reasoning': 'Critical risk override'
        }
    
    # P2: Strong Buy Signal
    if (trend_score >= 2 and risk_score <= 3 
        and forecast_change > 5):
        return {
            'decision': 'BUY',
            'confidence_score': 90,
            'decision_type': 'rule-based',
            'rule_triggered': 'P2',
            'reasoning': 'All indicators positive'
        }
    
    # P3, P4 rules...
    # (omitted for brevity)
    
    # LLM Fallback for ambiguous cases
    return self.llm_decide(trend_result, risk_result,
                           forecast_result)
\end{lstlisting}

\subsection*{C.2 LLM Evaluation Framework}

\begin{lstlisting}[language=Python, caption={Evaluation Core}]
def evaluate_comprehensive(self, agent_name, 
                          llm_response, ground_truth,
                          input_metrics, tokens_used):
    # Factual accuracy check
    accuracy = self.evaluate_factual_accuracy(
        llm_response, ground_truth
    )
    
    # Reasoning quality assessment
    quality = self.evaluate_reasoning_quality(
        llm_response, input_metrics
    )
    
    # Cost efficiency calculation
    cost = self.evaluate_cost_efficiency(
        llm_response, tokens_used
    )
    
    # Weighted overall score
    overall = (
        accuracy['accuracy_score'] * 0.35 +
        quality['quality_score'] * 0.35 +
        cost['cost_score'] * 0.30
    )
    
    return {
        'overall_score': overall,
        'dimensions': {
            'accuracy': accuracy,
            'quality': quality,
            'cost': cost
        }
    }
\end{lstlisting}

\subsection*{C.3 Parallel Agent Execution}

\begin{lstlisting}[language=Python, caption={Concurrent Analysis}]
from concurrent.futures import ThreadPoolExecutor

def analyze_stock(ticker):
    with ThreadPoolExecutor(max_workers=3) as executor:
        # Submit all agents concurrently
        trend_future = executor.submit(
            trend_agent.run, ticker
        )
        risk_future = executor.submit(
            risk_agent.run, ticker
        )
        forecast_future = executor.submit(
            forecast_agent.run, ticker
        )
        
        # Gather results
        trend_result = trend_future.result()
        risk_result = risk_future.result()
        forecast_result = forecast_future.result()
        
    # Decision agent runs sequentially
    decision = decision_agent.decide(
        trend_result, risk_result, forecast_result
    )
    
    return {
        'trend': trend_result,
        'risk': risk_result,
        'forecast': forecast_result,
        'decision': decision
    }
\end{lstlisting}
